df_video_count['date'] = pd.to_datetime(df_video_count['date'])

daily_stats = df_video_count.groupby('date').agg(total_views=('count', 'sum'),number_of_videos=('video_id', 'nunique')).reset_index()

daily_stats['avg_views_per_video'] = daily_stats['total_views'] / daily_stats['number_of_videos']

plt.figure(figsize=(15, 6))
plt.plot(daily_stats['date'], daily_stats['avg_views_per_video'])
plt.title('Total views/Number of videos')
plt.xlabel('date')
plt.ylabel('avg total views per video')
plt.grid(True)
plt.show()
